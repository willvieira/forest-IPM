pars_to_list <- function(pars) {
  pars |>
    select(contains('.')) |>
    pivot_longer(cols = everything()) |>
    mutate(
      vr  = str_replace(name, '\\..*', ''),
      par = str_replace(name, paste0(vr, '.'), ''),
      vr  = case_match(vr, 'recruit' ~ 'rec', .default = vr)
    ) |>
    select(!name) |>
    group_split(vr) |>
    set_names(map_chr(., ~.x$vr[1])) |>
    map(~.x |> select(!vr) |> pivot_wider(names_from = par) |> unlist(use.names = TRUE))
}

init_pop <- function(params, L = 127, h = 1, N = 0) {
  Lmax <- round(params[["growth"]]["Lmax"], 0)
  m    <- length(seq(L, Lmax, h))
  msh  <- L + ((seq_len(m)) - 0.5) * h
  if (N == 0) {
    return(list(meshpts = msh, Nvec = rep(0, m), h = h))
  }
  stop("N > 0 not needed for comparison script")
}

```r
scale_vars <- function(value, cov, direction, range_dt = vars_rg) {
  cov_rg  <- range_dt[[cov]]
  min_v   <- cov_rg[1]
  max_v   <- cov_rg[2]
  if (direction == 'scale')   return((value - min_v) / (max_v - min_v))
  if (direction == 'unscale') return(value * (max_v - min_v) + min_v)
  stop('`direction` must be "scale" or "unscale"')
}
```

7. **Per-row comparison function:**
   ```r
   compare_row <- function(array_id) {
     sim_pars  <- sim_pars_all[array_id, ]
     Sp        <- sim_pars$species_id
     plot_size <- sim_pars$plot_size
     temp      <- sim_pars$bio_01_mean
     prec      <- sim_pars$bio_12_mean

     set.seed(array_id)   # MUST be before any random call
     pop_pars <- pop_pars_all |>
       filter(species_id == Sp) |>
       select(!species_id) |>
       slice_sample(n = REPLICATIONS)

     plot_pars <- readRDS(file.path(SIM_DIR, "plot_parameters", paste0(Sp, ".RDS"))) |>
       filter(plot_id == sim_pars$plot_id, year_measured == sim_pars$year_measured) |>
       pull(plot_pars) |> .[[1]]

     ref_output <- final_output[[array_id]]
     temp_scl   <- scale_vars(temp, "temp", "scale")
     prec_scl   <- scale_vars(prec, "prec", "scale")

     metrics <- c("lambda_base", "par.BA_con", "par.BA_het", "par.temp", "par.prec")
     new_out  <- tibble(
       lambda_base = numeric(REPLICATIONS),
       par.BA_con  = numeric(REPLICATIONS),
       par.BA_het  = numeric(REPLICATIONS),
       par.temp    = numeric(REPLICATIONS),
       par.prec    = numeric(REPLICATIONS)
     )

     for (i in seq_len(REPLICATIONS)) {
       pop_pars_i <- pars_to_list(pop_pars[i, ])
       re_pars_i  <- plot_pars[i, ] |> unlist() |> unname()

       N_ref <- init_pop(pop_pars_i, L = 127, h = 1, N = 0)

       N_con <- dbh_to_sizeDist(dbh = unlist(sim_pars$con_dbh), N_intra = N_ref)

       if (is.null(unlist(sim_pars$het_dbh))) {
         N_het <- N_ref
       } else {
         N_het <- dbh_to_sizeDist(dbh = unlist(sim_pars$het_dbh), N_intra = N_ref)
       }

       mk_args <- list(
         Nvec_intra  = N_con,
         Nvec_inter  = N_het,
         delta_time  = 1,
         plotSize    = plot_size,
         Temp        = temp_scl,
         Prec        = prec_scl,
         pars        = pop_pars_i,
         plot_random = re_pars_i
       )
       lambda_base <- max(getEigenValues(do.call(mkKernel, mk_args)$K))

       # BA_con perturbation
       dbh_con_pertb <- scale_vars(
         scale_vars(unlist(sim_pars$con_dbh), "dbh", "scale") + PERTURBATION,
         "dbh", "unscale"
       )
       N_con_pertb <- dbh_to_sizeDist(dbh = dbh_con_pertb, N_intra = N_ref)
       lambda_bacon <- max(getEigenValues(
         mkKernel(N_con_pertb, N_het, 1, plot_size, temp_scl, prec_scl, pop_pars_i, re_pars_i)$K
       ))

       # BA_het perturbation
       if (is.null(unlist(sim_pars$het_dbh))) {
         N_het_pertb  <- N_het
         lambda_bahet <- lambda_base
       } else {
         dbh_het_pertb <- scale_vars(
           scale_vars(unlist(sim_pars$het_dbh), "dbh", "scale") + PERTURBATION,
           "dbh", "unscale"
         )
         N_het_pertb  <- dbh_to_sizeDist(dbh = dbh_het_pertb, N_intra = N_ref)
         lambda_bahet <- max(getEigenValues(
           mkKernel(N_con, N_het_pertb, 1, plot_size, temp_scl, prec_scl, pop_pars_i, re_pars_i)$K
         ))
       }

       # Temp perturbation
       lambda_temp <- max(getEigenValues(
         mkKernel(N_con, N_het, 1, plot_size, temp_scl + PERTURBATION, prec_scl, pop_pars_i, re_pars_i)$K
       ))

       # Prec perturbation
       lambda_prec <- max(getEigenValues(
         mkKernel(N_con, N_het, 1, plot_size, temp_scl, prec_scl + PERTURBATION, pop_pars_i, re_pars_i)$K
       ))

       new_out$lambda_base[i] <- lambda_base
       new_out$par.BA_con[i]  <- abs(lambda_bacon - lambda_base) /
                                  (size_to_BAplot(N = N_con_pertb, plot_size) - size_to_BAplot(N = N_con, plot_size))
       new_out$par.BA_het[i]  <- abs(lambda_bahet - lambda_base) /
                                  (size_to_BAplot(N = N_het_pertb, plot_size) - size_to_BAplot(N = N_het, plot_size))
       new_out$par.temp[i]    <- abs(lambda_temp - lambda_base) / PERTURBATION
       new_out$par.prec[i]    <- abs(lambda_prec - lambda_base) / PERTURBATION
     }

     diffs <- sapply(metrics, function(m) max(abs(new_out[[m]] - ref_output[[m]])))
     list(
       array_id  = array_id,
       species   = Sp,
       het_null  = is.null(unlist(sim_pars$het_dbh)),
       pass      = all(diffs < TOLERANCE),
       max_diffs = diffs
     )
   }
   ```

8. **Main execution and results table:**

   cat(sprintf("\nRunning comparison for %d selected rows...\n", length(selected_rows)))
   results <- lapply(selected_rows, function(id) {
     cat(sprintf("  Row %6d ... ", id))
     res <- tryCatch(compare_row(id), error = function(e) list(
       array_id = id, species = NA, het_null = NA, pass = FALSE,

     het_null = sapply(results, `[[`, "het_null"),

     }
   }
