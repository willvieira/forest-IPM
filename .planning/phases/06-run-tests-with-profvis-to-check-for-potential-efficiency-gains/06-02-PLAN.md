    ```r
    # profile_benchmark.R
    # Standalone profiling script — not shipped with the package.
    # Run from the package root after devtools::load_all().
    # Requires: profvis, microbenchmark (install.packages if missing)

    library(forestIPM)
    library(profvis)
    library(microbenchmark)

    # ── Benchmark inputs ──────────────────────────────────────────────────────────

    # Single-species: QUERUB (large max DBH — stresses outer() kernel build)
    df_single <- data.frame(
      size_mm    = seq(130, 600, by = 10),  # ~48 individuals across size range
      species_id = "QUERUB",
      plot_size  = 1000
    )
    s_single   <- stand(df_single)
    mod_single <- species_model(s_single, species = "QUERUB")
    pars_single <- parameters(mod_single, draw = "mean")
    env        <- env_condition(MAT = 10, MAP = 1100)
    ctrl_100   <- control(years = 100, compute_lambda = FALSE, progress = FALSE)

    # Multi-species: QUERUB + QUEPRI (exposes competition update cost)
    df_multi <- data.frame(
      size_mm    = c(seq(130, 600, by = 10), seq(130, 400, by = 10)),
      species_id = c(rep("QUERUB", length(seq(130, 600, by = 10))),
                     rep("QUEPRI", length(seq(130, 400, by = 10)))),
      plot_size  = 1000
    )
    s_multi    <- stand(df_multi)
    mod_multi  <- species_model(s_multi, species = c("QUERUB", "QUEPRI"))
    pars_multi <- parameters(mod_multi, draw = "mean")
    ctrl_multi <- control(years = 100, compute_lambda = FALSE, progress = FALSE)

    # Log mesh size so PROFILING.md can document it
    message(sprintf("QUERUB mesh size: %d points", length(s_single$data$size_mm)))

    # ── profvis: project() single-species ─────────────────────────────────────────
    p_project_single <- profvis({
      project(mod_single, pars_single, s_single, env, ctrl_100)
    }, interval = 0.005, height = "600px")

    # ── profvis: lambda() single kernel build path ────────────────────────────────
    ctrl_1 <- control(years = 1, compute_lambda = TRUE, progress = FALSE)
    p_lambda <- profvis({
      for (i in 1:200) lambda(mod_single, pars_single, s_single, env, ctrl_1)
    }, interval = 0.005, height = "600px")

    # ── profvis: project() multi-species ──────────────────────────────────────────
    p_project_multi <- profvis({
      project(mod_multi, pars_multi, s_multi, env, ctrl_multi)
    }, interval = 0.005, height = "600px")

    # ── Save flamegraphs ──────────────────────────────────────────────────────────
    phase_dir <- ".planning/phases/06-run-tests-with-profvis-to-check-for-potential-efficiency-gains"
    htmlwidgets::saveWidget(
      p_project_single,
      file.path(phase_dir, "profvis_output.html"),
      selfcontained = TRUE
    )
    # Also save multi-species and lambda profiles for reference
    htmlwidgets::saveWidget(
      p_project_multi,
      file.path(phase_dir, "profvis_multi_output.html"),
      selfcontained = TRUE
    )

    # ── microbenchmark: mkKernel() alone ─────────────────────────────────────────
    # Load internals to benchmark mkKernel directly
    # (mkKernel is not exported but accessible via devtools::load_all())
    nvec_list <- forestIPM:::.stand_to_nvec(s_single, "QUERUB", pars_single, ctrl_1$bin_width)
    sp_pars   <- pars_single$species_params[["QUERUB"]]$fixed

    mb_kernel <- microbenchmark(
      mkKernel_call = {
        mkKernel(
          Nvec_intra  = nvec_list[["QUERUB"]]$N_con,
          Nvec_inter  = nvec_list[["QUERUB"]]$N_het,
          delta_time  = ctrl_1$delta_time,
          plotSize    = s_single$plot_size,
          Temp        = env$.MAT_scl,
          Prec        = env$.MAP_scl,
          pars        = sp_pars,
          plot_random = c(0, 0, 0)
        )
      },
      times = 50
    )
    print(mb_kernel)

    # Print profvis summaries to console
    message("\n=== Single-species project() profile ===")
    print(p_project_single$x$message)
    ```
