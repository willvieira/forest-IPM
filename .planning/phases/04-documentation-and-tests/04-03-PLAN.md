New forestIPM package API (replaces old procedural code):
```r
library(forestIPM)

# Constructor chain
s    <- stand(data.frame(size_mm = ..., species_id = "ABIBAL", plot_size = 1000))
mod  <- species_model(s)
pars <- parameters(mod, draw = "mean")        # or draw = 1L for posterior sample
env  <- env_condition(MAT = 8, MAP = 1200)   # temperate climate example
ctrl <- control(years = 100, compute_lambda = TRUE, progress = TRUE)

# Engines
lam  <- lambda(mod, pars, s, env, ctrl)       # returns ipm_lambda
proj <- project(mod, pars, s, env, ctrl)      # returns ipm_projection

# Visualization (three-panel idiom — NEW in Plan 01)
plot(proj)                        # all three figures
plot(proj, type = "lambda")       # lambda trajectory
plot(proj, type = "size_dist")    # size distribution over time
plot(proj, type = "lambda_vs_n")  # lambda vs N correlation

# Species discovery
supported_species()               # returns data frame of 31 species
```
